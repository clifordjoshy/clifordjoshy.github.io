{{/* An image shortcode which handles centering and sizing This overrides the theme shortcode (that adds lazysizes and lightgallery) It does make use of the loading.svg from the theme though. */}} {{- $src := cond .IsNamedParams (.Get "src") (.Get 0) -}} {{- $caption := cond .IsNamedParams (.Get "caption") (.Get 1) -}} {{- $width := or (.Get "width") "fit-content" -}} {{- $height := or (.Get "height") "fit-content" -}} {{- $alt := (.Get "alt") -}} {{- $href := (.Get "href") -}} {{- $loading := resources.Get "svg/loading.svg" | minify -}} {{- if not (.Page.Scratch.Get "imgLoaderJSEmitted") -}} {{- .Page.Scratch.Set "imgLoaderJSEmitted" true -}} {{- end -}} {{- $targetPath := $src -}} {{- if not (or (hasPrefix $src "http://") (hasPrefix $src "https://") (hasPrefix $src "/")) -}} {{- $targetPath = printf "/%s" (path.Clean (path.Join (path.Dir .Page.File.Path) $src)) -}} {{- end -}} {{- if eq $targetPath "" -}} {{- errorf "invalid image source: %s. Can't resolve path %s" $src $targetPath -}} {{- end -}} {{- if $caption -}} {{- end -}}
