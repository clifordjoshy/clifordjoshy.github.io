{{/* An image shortcode which handles centering and sizing This overrides the theme shortcode (that adds lazysizes and lightgallery) We generate a full URL of the image instead of a relative path, so that RSS readers can render the image correctly. It does make use of the loading.svg from the theme though. */}} {{- $src := cond .IsNamedParams (.Get "src") (.Get 0) -}} {{- $caption := cond .IsNamedParams (.Get "caption") (.Get 1) -}} {{- $width := or (.Get "width") "fit-content" -}} {{- $height := or (.Get "height") "fit-content" -}} {{- $alt := (.Get "alt") -}} {{- $href := (.Get "href") -}} {{- $loading := resources.Get "svg/loading.svg" | minify -}} {{- if not (.Page.Scratch.Get "imgLoaderJSEmitted") -}} {{- .Page.Scratch.Set "imgLoaderJSEmitted" true -}} {{- end -}} {{- $url := "" -}} {{- if or (hasPrefix $src "http://") (hasPrefix $src "https://") (hasPrefix $src "/" -}} {{- $url = $src -}} {{- else -}} {{- $url = path.Clean (path.Join (path.Dir .Page.RelPermalink) $src) -}} {{- end -}} {{- if eq $url "" -}} {{- errorf "invalid image source: %s" $src -}} {{- end -}} {{- if $caption -}} {{- end -}}
